# Para subir un nuevo repositorio
- echo "# example" >> README.md
- git init
- git add README.md
- git commit -m "first commit"
- git branch -M main
- git remote add origin https://github.com/samuelbonet/example.git
- git push -u origin main

# Para actualizar el repositorio
- git remote add origin https://github.com/samuelbonet/example.git
- git branch -M main
- git push -u origin main
